An example project from beginning to end¶

This is an example project including the cleaning of a (very) messy data source, some data exploration, and a simple learning model. You can use it as a reference for your own end-of-the-term project.

The objective is to forecast covid case numbers given past vaccine / positive case / demographic data in LA counties.

This is a typical case of real-world data collection from various sources with no guarantee of data consistency and quality. The data collection process is also unknown to the data scientist, so you have to first of all understand the data structure and make some educated guesses, sometimes with external data to double check.

def normal_test(df):
    # normality test
    stat, p = shapiro(df)
    print('Statistics=%.3f, p=%.3f' % (stat, p))
    # interpret
    alpha = 0.05
    if p > alpha:
        print('Sample looks Gaussian (fail to reject H0)')
    else:
        print('Sample does not look Gaussian (reject H0)')
    return p>alpha, np.mean(df), np.var(df)

def independent_ttest(data1, data2):
    isnormal1, mu1, var1 = normal_test(data1)
    isnormal2, mu2, var2 = normal_test(data2)
    print('Sample1 normal_test: {}, sample2 normal_test: {}'.format(isnormal1, isnormal2))
    equal_var = False
    if abs(var1 - var2) / max(var1, var2) < 0.1:
        equal_var = True
    res = ttest_ind(a=data1, b=data2, equal_var=equal_var)
    alpha = 0.05
    if res.pvalue > alpha:
        print('Sample means do NOT differ (fail to reject H0)')
    else:
        print('Sample means are different (reject H0)')

def merge_data(vac, cas, interval=None, county=None):
    # vac = vac[vac['county'] != 'Statewide']
    vac['demography'] = vac['demographic_category'] + '_' + vac['demographic_value']
    white_perc_wiki = vac.loc[~vac['White_perc_wiki'].isnull(), ['county', 'White_perc_wiki']].groupby(
        'county', as_index=False).max()
    cols = ['county', 'date', 'demography', 'partially_vaccinated', 'total_partially_vaccinated',
            'fully_vaccinated', 'cumulative_fully_vaccinated', 'cumulative_unvax_total_pop',
            'cumulative_booster_recip_count', 'est_population_final']
    vac = vac[cols]
    # consolidate numbers in time intervals
    if interval is not None:
        interval = 86400 * interval  # date format is seconds from epoch time
        vac['date'] = vac['date'].apply(lambda x: int(x / interval) * interval)
        vac = vac.groupby(['county', 'date', 'demography'], as_index=False).agg({
            'partially_vaccinated': 'sum', 'total_partially_vaccinated': 'max', 'fully_vaccinated': 'sum',
            'cumulative_fully_vaccinated': 'max', 'cumulative_unvax_total_pop': 'min',
            'cumulative_booster_recip_count': 'max', 'est_population_final': 'mean'})
    vac = vac.pivot_table(index=['county', 'date'], columns='demography', values=[
            'partially_vaccinated', 'total_partially_vaccinated', 'fully_vaccinated', 'cumulative_fully_vaccinated',
            'cumulative_unvax_total_pop', 'cumulative_booster_recip_count', 'est_population_final'], aggfunc='mean')
    vac.columns = ['_'.join(col).strip() for col in vac.columns.values]
    vac.reset_index(inplace=True)
    vac = vac.merge(white_perc_wiki, on='county', how='left')
    vac = vac.merge(cas, on=['county', 'date'], how='left')
    if county is not None:
        vac = vac[vac['county'] == county]
        vac.drop('county', axis=1, inplace=True)
        vac.fillna(0, inplace=True)
        vac.sort_values(by='date', ascending=True, inplace=True)
        return vac
    county_dummies = pd.get_dummies(vac['county'])
    vac = pd.concat([county_dummies, vac[vac.columns[1:]]], axis=1)
    vac.fillna(0, inplace=True)
    vac.sort_values(by=['county', 'date'], ascending=True, inplace=True)
    return vac

import torch
from torch import nn, optim, tensor
torch.set_num_threads(1)
DEVICE = torch.device('cpu')
DTYPE = torch.FloatTensor


class LSTM(nn.Module):
    hidden_size = 128

    def __init__(self, input_size, output_size=1):
        super().__init__()
        self.input_size = input_size
        self.output_size = output_size

        self.batchNorm = nn.BatchNorm1d(input_size)
        self.lstm = nn.LSTM(input_size, self.hidden_size)
        self.output = nn.Linear(self.hidden_size, output_size)

    def forward(self, record):
        batch = self.batchNorm(record)
        lstm_out, _ = self.lstm(batch.view(len(batch), 1, -1))
        output = self.output(lstm_out.view(len(batch), -1))
        return output


class Net(nn.Module):
    epoch = 20
    criterion = nn.MSELoss()

    def __init__(self, batch_size, input_size, output_size=1,path=None):
        super().__init__()
        self.model = LSTM(input_size=input_size, output_size=output_size)
        if path is not None:
            self.path = 'model.pkl'
            self._reload_model()
        self.model.to(DEVICE)
        self.batch_size = batch_size
        self.loss_function = self.criterion
        self.optimizer = optim.SGD(self.model.parameters(), lr=0.1)

    def train(self, training_data, target_col):
        self.epoch_loss = []
        for epoch in range(self.epoch):
            epoch_loss = []
            records = np.random.choice(training_data.index, size=len(training_data), replace=False)
            for rec in records:
                input_data = training_data.loc[rec - self.batch_size + 1: rec]
                if len(input_data) < self.batch_size:
                    continue
                target_data = tensor(np.array(input_data[target_col]), device=DEVICE).type(DTYPE)
                target_data = target_data.view(-1, 1)
                input_data = tensor(np.array(input_data.drop(target_col, axis=1)), device=DEVICE).type(DTYPE)
                self.model.zero_grad()
                estim_output = self.model(input_data)
                loss = self.loss_function(estim_output, target_data)
                epoch_loss.append(loss.detach().numpy())
                loss.backward()
                self.optimizer.step()
            print('epoch {}: {}'.format(epoch, sum(epoch_loss) / len(epoch_loss)))
            self.epoch_loss.append(sum(epoch_loss) / len(epoch_loss))
        self._save_model('model.pkl')

    def infer(self, test_data):
        # See what the scores are after training
        with torch.no_grad():
            nr_rec = len(test_data) - self.batch_size + 1
            estim_output = None
            for i in range(nr_rec):
                input_data = test_data.iloc[i: i + self.batch_size]
                if len(input_data) < self.batch_size:
                    continue
                input_data = tensor(np.array(input_data), device=DEVICE).type(DTYPE)
                if estim_output is None:
                    estim_output = self.model(input_data).detach().numpy()
                else:
                    estim_output = np.vstack([estim_output, self.model(input_data).detach().numpy()[-1]])

def normalize(df, mean=None, denom=None, log=None):
    if log is None:
        isnormal = normal_test(df)
        log = not(isnormal[0])
    if log:
        df = np.where(df==0, 1e-2, df)
        df = np.log(df)
        # sns.histplot(df, kde=True)
    if mean is None:
        mean = np.mean(df)
    if denom is None:
        denom = np.std(df)
    df = (df - mean) / denom
    return df, mean, denom, log


def inverse_transform(df, mean, denom, log):
    res = df * denom + mean
    if log:
        return np.exp(res).astype(int)
    return res.astype(int)


def regress(df, target_col='deaths', batch_size=15):
    target_column, mean, denom, log = normalize(df[target_col])
    df[target_col] = target_column
    test_data_perc = 0.1
    test_data_length = int(len(df) * test_data_perc) + batch_size
    training_data = df.iloc[0:-test_data_length]
    test_data = df.iloc[-test_data_length:]
    test_input = test_data.drop(target_col, axis=1)
    test_target = np.array(test_data[target_col][-(test_data_length-batch_size+1):])
    test_target = test_target.reshape(-1, 1)
    mdl = Net(batch_size=batch_size, input_size=df.shape[1] - 1, output_size=1)
    mdl.train(training_data, target_col=target_col)
    res = mdl.infer(test_data=test_input)[-(test_data_length-batch_size+1):]
    loss = sum((res - test_target) ** 2) / len(res)
    res = inverse_transform(res, mean, denom, log)
    return mdl, res, loss, mean, denom, log
